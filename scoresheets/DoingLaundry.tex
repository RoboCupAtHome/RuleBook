\begin{scorelist}[timelimit=7]
    \scoreheading{Main Goal}
    \scoreitem{15}{Navigating to the laundry area}
    \scoreitem{100}{Picking up a piece of clothing from the basket}
      \scorepen{-100}{Picking up multiple at once}
    \scoreitem{800}{Folding a piece of clothing}
      \scorepen{200}{%
        Human assistance: Flattening/Arrange clothing before folding
      }
      \scorepen{800}{%
        Human assistance during folding will apply a \\
        \hspace{3em}percentage penalty according to the amount of help
      }

    \scoreheading{Extra Rewards}
    \scoreitem{300}{Opening the washing machine door}
    \scoreitem{300}{Removing one or more piece of clothing from the washing machine}
      \scorepen{200}{%
        The clothing touches the floor
      }
    \scoreitem{300}{Using the basket for transportation}
      \scorepen{200}{%
        Laundry is dropped or otherwise lost during transportation.
      }
    \scoreitem[5]{400}{Folding additional clothes (per item)}
      \scorepen{200}{%
        Human assistance: Flattening/Arrange clothing before folding
      }
      \scorepen[5]{400}{%
          Human assistance during folding will apply a \\
          \hspace{3em}percentage penalty according to the amount of help
        }
    \scoreitem[6]{100}{Stacking folded clothes neatly (per item)}

\end{scorelist}
